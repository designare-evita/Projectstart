// js/ai-form.js import { startPlaceholderAnimation, stopPlaceholderAnimation } from './typewriter.js'; export function initAiForm() { const aiForm = document.getElementById('ai-form'); if (!aiForm) return; // Alle DOM-Elemente holen const aiQuestionInput = document.getElementById('ai-question'); const aiStatus = document.getElementById('ai-status'); const submitButton = aiForm.querySelector('button[type="submit"]'); const modal = document.getElementById('ai-response-modal'); if (!modal) { console.error("Das AI-Response-Modal wurde nicht im DOM gefunden."); return; } const responseContent = document.getElementById('ai-response-content'); const startChatButton = document.getElementById('start-chat-button'); const closeModalButton = document.getElementById('close-modal-button'); const initialResponseContainer = document.getElementById('initial-ai-response'); const chatContainer = document.getElementById('ai-chat-container'); const chatHistory = document.getElementById('chat-history'); const chatForm = document.getElementById('chat-form'); const chatInput = document.getElementById('chat-input'); // ----- HILFSFUNKTIONEN FÜR DAS MODAL ----- const openModal = () => { modal.classList.add('visible'); document.body.classList.add('no-scroll'); // Verhindert das Scrollen des Hintergrunds }; const closeModal = () => { modal.classList.remove('visible'); document.body.classList.remove('no-scroll'); }; // ----- HAUPTLOGIK ----- // Formular-Absendung für die ERSTE Frage aiForm.addEventListener('submit', async (e) => { e.preventDefault(); const question = aiQuestionInput.value.trim(); if (!question) return; stopPlaceholderAnimation(); aiStatus.innerText = "Einen Moment, Evita gleicht gerade ihre Bits und Bytes ab..."; aiStatus.classList.add('thinking'); aiQuestionInput.disabled = true; submitButton.disabled = true; try { const response = await fetch('/api/ask-gemini', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ question: question }) }); if (!response.ok) { const errorData = await response.json().catch(() => ({ error: 'Netzwerk-Antwort war nicht OK.' })); throw new Error(errorData.error); } const data = await response.json(); const aiAnswer = data.answer || "Ich habe leider keine Antwort erhalten."; // Antwort ins Modal füllen responseContent.innerHTML = aiAnswer; // Modal im initialen Zustand anzeigen initialResponseContainer.style.display = 'block'; chatContainer.style.display = 'none'; closeModalButton.style.display = 'none'; openModal(); // Korrekte Funktion zum Öffnen verwenden! // Event-Listener für "Frag Evita"-Button (wird hier gesetzt, da aiAnswer bekannt sein muss) startChatButton.onclick = function() { initialResponseContainer.style.display = 'none'; chatContainer.style.display = 'block'; closeModalButton.style.display = 'block'; chatHistory.innerHTML = `
${aiAnswer}
`; chatInput.focus(); }; } catch (error) { console.error("Fehler bei der KI-Anfrage:", error); aiStatus.innerText = `Fehler: ${error.message}`; } finally { aiQuestionInput.value = ''; aiQuestionInput.disabled = false; submitButton.disabled = false; aiStatus.classList.remove('thinking'); if (!aiStatus.innerText.startsWith('Fehler')) { aiStatus.innerText = ''; } startPlaceholderAnimation(); } }); // Formular-Absendung für den Chat chatForm.onsubmit = async function(event) { event.preventDefault(); const userMessage = chatInput.value.trim(); if (!userMessage) return; chatHistory.innerHTML += `
${userMessage}
`; chatInput.value = ''; chatHistory.scrollTop = chatHistory.scrollHeight; const thinkingElement = document.createElement('div'); thinkingElement.className = 'ai-message thinking'; thinkingElement.innerHTML = '... tippt ...'; chatHistory.appendChild(thinkingElement); chatHistory.scrollTop = chatHistory.scrollHeight; try { const response = await fetch('/api/ask-gemini', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ question: userMessage, context: chatHistory.innerText }) }); if (!response.ok) { const errorData = await response.json().catch(() => ({ error: 'Chat-Antwort war nicht OK.' })); throw new Error(errorData.error); } const data = await response.json(); thinkingElement.innerHTML = data.answer || "Ich konnte keine Antwort finden."; thinkingElement.classList.remove('thinking'); } catch (error) { thinkingElement.innerHTML = `Entschuldigung, ein Fehler ist aufgetreten.`; thinkingElement.classList.add('error-message'); console.error('Error fetching AI chat response:', error); } finally { chatHistory.scrollTop = chatHistory.scrollHeight; } }; // Schließen des Modals closeModalButton.onclick = closeModal; modal.addEventListener('click', (e) => { if (e.target === modal) { closeModal(); } }); }
